# variables
rg="<resource-group>"
location="<location>"
old="id-set-dev-tcuv"
new="id-set-dev-tcuv-a"
kv_name="<keyvault-name>"

# create new identity
az identity create -g "$rg" -n "$new" -l "$location"

# get principalId for RBAC assignment
new_principal_id="$(az identity show -g "$rg" -n "$new" --query principalId -o tsv)"

# assign Key Vault role (RBAC model)
kv_id="$(az keyvault show -g "$rg" -n "$kv_name" --query id -o tsv)"

az role assignment create \
  --assignee-object-id "$new_principal_id" \
  --assignee-principal-type ServicePrincipal \
  --role "Key Vault Secrets User" \
  --scope "$kv_id"
